

find . -type d \( -name ".gitignore" -o -name ".next" -o -name "node_modules" -o -name ".env" \) -prune -o -type f ! -name "code_complet_projet.txt" -print | while read -r fichier; do
    # On vérifie que c'est bien un fichier texte (pour ignorer les images, PDF, etc.)
    if file "$fichier" | grep -q "text"; then
        echo -e "\n============================================================"
        echo "Fichier : $fichier"
        echo -e "============================================================\n"
        cat "$fichier"
    fi
done > code_complet_projet.txt
